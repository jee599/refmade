api-example.ts
TypeScript 
import { TaskFlow } from '@taskflow/sdk'

const tf = new TaskFlow({
  apiKey: process.env.TASKFLOW_KEY,
  ai: 'auto'
})

// Create sprint from GitHub milestone
const sprint = await tf.createSprint({
  source: 'github',
  milestone: 'v2.1',
  autoAssign: true,
  optimize: 'velocity'
})

// AI distributes by capacity + history
const plan = await sprint.optimize()

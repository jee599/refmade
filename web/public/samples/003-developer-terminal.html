Use our client libraries to get started with your favorite framework. DevBase handles the infrastructure so you can focus on building your product. 

// Initialize the client
import { createClient } from '@devbase/devbase-js'

const devbase = createClient(
  'https://xyzcompany.devbase.co',
  'public-anon-key'
)

// Fetch data with realtime
const { data, error } = await devbase
  .from('countries')
  .select('*')
  .order('name', { ascending: true })
  .limit(10)

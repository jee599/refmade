Quick start

Deploy in under 60 seconds

# Install the CLI
npx @graphnode/cli init
 
# Add your subgraphs
import { defineConfig } from '@graphnode/sdk'
 
export default defineConfig({
  subgraphs: [
    { name: 'users', url: 'https://api.example.com/users' },
    { name: 'products', url: 'https://api.example.com/products' },
  ],
  governance: { auth: 'rbac', rateLimit: '1000/min' },
})
 
# Deploy to the edge
npx graphnode deploy --prod
